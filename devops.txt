pipeline {
agent any
environment {
PROJECT_DIR = "/home/student/shemeema/hello-world"
}
stages {
stage('Build') {
steps {
dir("${PROJECT_DIR}") {
sh '
export PATH=$HOME/maven/bin:$PATH
mvn clean package
'
}
}
}
stage('Verify') {
steps {
sh 'ls-l /home/student/shemeema/hello-world/target/'
}
}
stage('Deploy') {
steps {
sh '
export LC_ALL=C.UTF-8
export LANG=C.UTF-8
export PATH=$HOME/.local/bin:$PATH
ansible-playbook-i /home/student/shemeema/hosts.ini
/home/student/shemeema/setup.yml
'
}
}
}
post {
success {
echo 'BUILD + DEPLOYSUCCESS✅'
}
failure {
echo 'PIPELINE FAILED ❌'
}
